    <header>

        <div id="data-panel">

            Latitude: <span id="lat">0.00</span>° | 

            Longitude: <span id="lon">0.00</span>° | 

            Velocity: <span id="vel">0</span> km/h

        </div>

    </header>


    <!-- Map container -->

    <div id="map"></div>


    <!-- Leaflet JavaScript -->

    <script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>

    

    <script>

        // 1. Initialize the Map centered at (0,0) with a low zoom level

        const map = L.map('map').setView([0, 0], 2);


        // 2. Add OpenStreetMap tiles to our map

        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {

            maxZoom: 18,

            attribution: '© OpenStreetMap contributors'

        }).addTo(map);


        // 3. Create a custom icon for the ISS

        const issIcon = L.icon({

            iconUrl: 'https://upload.wikimedia.org/wikipedia/commons/d/d0/International_Space_Station.svg',

            iconSize: [50, 30], // size of the icon

            iconAnchor: [25, 15] // point of the icon which will correspond to marker's location

        });


        // 4. Create the marker and add it to the map

        const marker = L.marker([0, 0], { icon: issIcon }).addTo(map);


        // URL for the ISS API

        const api_url = 'https://api.wheretheiss.at/v1/satellites/25544';


        let firstLoad = true;


        // 5. Fetch ISS data from the API

        async function getISS() {

            try {

                const response = await fetch(api_url);

                const data = await response.json();

                

                // Extract coordinates and velocity

                const { latitude, longitude, velocity } = data;


                // Update the marker position

                marker.setLatLng([latitude, longitude]);

                

                // If it's the first time loading, center the map view on the ISS

                if (firstLoad) {

                    map.setView([latitude, longitude], 3);

                    firstLoad = false;

                }


                // Update the HTML text elements (rounded to 4 decimal places)

                document.getElementById('lat').textContent = latitude.toFixed(4);

                document.getElementById('lon').textContent = longitude.toFixed(4);

                document.getElementById('vel').textContent = Math.round(velocity).toLocaleString();


            } catch (error) {

                console.error("Error fetching ISS data:", error);

            }

        }


        // Call the function immediately on load

        getISS();


        // Refresh the data every 2 seconds (2000 milliseconds)

        setInterval(getISS, 2000);

    </script>
